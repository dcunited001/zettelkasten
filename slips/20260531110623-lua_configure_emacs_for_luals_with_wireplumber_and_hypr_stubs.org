*** Where to put the =stubs=?

#+begin_src shell
# locate '/data/ecto/x.files/*/*.lua' \
fd -Hp '/data/ecto/x.files/.*\.lua' \
    | sed -E 's/(.*)\/.*\.lua$/\1/g' | sort | uniq \
