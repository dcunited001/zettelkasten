#+begin_src emacs-lisp
(setq listy-list '(1 2 3 4 5)
      it-scalar 5
      it-string "asdf")

(-> it-string
    reverse
    (concat it-string))


(-> it-scalar
    (+ 10)
    (+))

;; thread-first
(--> it-scalar
     (* it it))

(--> it-string
     (concat it-string)
     reverse
     )

;; thread-more
;; (-> it-scalar
;;     (--> ...))

;; macro-expand it

;; a hash of squares
(->> listy-list
     (--annotate (* it it)))

;; a hash of cubes
(->> listy-list
     (--annotate (* it it))
     (--reduce (-lambda (* (assoc it) (rassoc it)))))

#+end_src
